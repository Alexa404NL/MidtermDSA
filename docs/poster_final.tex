\begin{block}{I. Project Overview}

\vspace{0.3em}
$\bullet$ Finds corrections based on minimum edits

\vspace{0.4em}
$\bullet$ 5-dimensional word embeddings

\vspace{0.3em}

\begin{block}{III. Problem Statement}
\textbf{Context:} Spell checking is fundamental in text editors, search engines, and messaging platforms.

\vspace{1em}
\textbf{Challenges:}

\vspace{0.4em}
$\bullet$ Finding similar words efficiently

\vspace{0.3em}

\vspace{1.2em}
\textbf{Formal Definition:}

\vspace{0.5em}
Given dictionary $D$ and input word $q$:

\vspace{0.6em}
\textit{A* + Trie:}
\[ w^* = \arg\min_{w \in D} \text{Lev}(q, w) \]

\vspace{0.3em}
\textit{KD-Tree:}
\[ w^* = \arg\min_{w \in D} \|\text{embed}(q) - \text{embed}(w)\|_2 \]

\begin{block}{IV. Algorithm Design}
\textcolor{triecolor}{\textbf{\large Algorithm 1: A* Search in Trie}}

\vspace{0.8em}
\textbf{Trie Structure:}

\vspace{0.4em}
$\bullet$ Prefix tree for efficient storage

\vspace{0.3em}
$\bullet$ Each node: children map + isEndOfWord flag

\vspace{0.8em}
\textbf{A* Search Function:}
\[ f(n) = g(n) + h(n) \]
where $g(n)$ = accumulated Levenshtein distance and $h(n)$ = heuristic estimate.

\vspace{0.8em}
\textbf{Levenshtein Distance:}
\[ d(i,j) = \min \begin{cases}
  d(i-1,j) + 1 & \text{(del)}\\
  d(i,j-1) + 1 & \text{(ins)}\\
  d(i-1,j-1) + c & \text{(sub)}
\end{cases} \]

\vspace{1.2em}
\begin{center}\color{border}\rule{0.9\linewidth}{1pt}\end{center}
\vspace{1em}

\textcolor{kdcolor}{\textbf{\large Algorithm 2: KD-Tree Search}}

\vspace{0.8em}
\textbf{5-Dimensional Word Embedding:}

\vspace{0.4em}
\begin{enumerate}
\item Normalized length: $\text{len}(w)/20$
\item Vowel frequency
\item Consonant frequency
\item First character normalized
\item Last character normalized
\end{enumerate}

\vspace{0.5em}
\textbf{Euclidean Distance:}
\[ d(\vec{p}, \vec{q}) = \sqrt{\sum_{i=1}^{5} (p_i - q_i)^2} \]

\vspace{1.2em}
\begin{center}\color{border}\rule{0.9\linewidth}{1pt}\end{center}
\vspace{1em}

\textcolor{parallelcolor}{\textbf{\large OpenMP Parallelization}}

\vspace{0.6em}
$\bullet$ Text split into word chunks

\vspace{0.3em}
$\bullet$ Dynamic scheduling for load balancing

\vspace{0.3em}
$\bullet$ \texttt{\#pragma omp parallel for schedule(dynamic)}

\vspace{0.8em}
\textbf{Parallel Levenshtein (Anti-diagonal):}

\vspace{0.4em}
$\bullet$ DP matrix computed by anti-diagonals

\vspace{0.3em}
$\bullet$ Cells in same diagonal are independent

\vspace{0.3em}
$\bullet$ Best for batch processing: \textbf{5.42x speedup}
\end{block}

\begin{block}{V. Complexity Analysis}
\textbf{Time Complexity:}

\vspace{0.6em}
\begin{center}
\begin{tabular}{lcc}
  \toprule
  \textbf{Operation} & \textbf{A* + Trie} & \textbf{KD-Tree} \\
  \midrule
  Construction & $O(n \cdot m)$ & $O(n \log n)$ \\
  Search (Best) & $O(m)$ & $O(\log n)$ \\
  Search (Avg) & $O(m \cdot d)$ & $O(\log n)$ \\
  Search (Worst) & $O(n \cdot m)$ & $O(n)$ \\
  \bottomrule
\end{tabular}
\end{center}

\vspace{0.4em}
{\small $n$ = dict size, $m$ = avg word length, $d$ = max edit dist}

\vspace{1.2em}
\textbf{Space Complexity:}

\vspace{0.6em}
\begin{center}
\begin{tabular}{lcc}
  \toprule
  \textbf{Structure} & \textbf{A* + Trie} & \textbf{KD-Tree} \\
  \midrule
  Dictionary & $O(n \cdot m)$ & $O(n \cdot k)$ \\
  Search Space & $O(m \cdot |\Sigma|)$ & $O(\log n)$ \\
  \bottomrule
\end{tabular}
\end{center}

\vspace{0.4em}
{\small $|\Sigma|$ = 26 (alphabet), $k$ = 5 (dimensions)}

\vspace{1.2em}
\textbf{Parallel Speedup (Amdahl's Law):}
\[ S(p) = \frac{1}{(1-f) + \frac{f}{p}} \]

With $f \approx 0.85$ and $p = 12$: $S_{\text{theoretical}} = 4.7$, $S_{\text{actual}} = 3.53$ (75\% efficiency)
\end{block}
